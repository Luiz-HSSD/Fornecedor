/*
drop TRIGGER set_for;
drop table fornecedor;
drop SEQUENCE forne;
*/
CREATE SEQUENCE forne START WITH 1;
create table fornecedor
(
forne_id NUMBER(13) primary key,
fornecedor_nome VARCHAR2(40),
cnpj char(14)
);

CREATE OR REPLACE TRIGGER set_for
BEFORE INSERT ON fornecedor 
FOR EACH ROW
BEGIN
  SELECT forne.NEXTVAL
  INTO   :new.forne_id
  FROM   dual;
END set_cat;
